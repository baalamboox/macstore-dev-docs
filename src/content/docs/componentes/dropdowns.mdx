/* Redimencionamiento del Nav para diferentes resoluciones de pantalla */
function agregarclase() {
    var elementoer = document.getElementById("bs-example-navbar-collapse-1");
    if (window.innerWidth >= 1025) {
      elementoer.classList.add("width2");
    } else {
      elementoer.classList.remove("width2");
    }
}

window.addEventListener("load", agregarclase);
window.addEventListener("resize", agregarclase);

const nav = document.querySelector("#hamburger button");
var box = document.getElementById('nav');

/* Abrir y cerrar el navbar */
nav.addEventListener("click", e => {
    nav.classList.toggle('open');
    box.classList.toggle('stykynav');
});
